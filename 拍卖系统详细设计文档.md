# 拍卖系统详细设计文档

## 1. 系统概述

### 1.1 项目背景

设计一个在线拍卖系统，支持用户参与竞拍、发布拍品、实时竞价等功能。

### 1.2 技术架构

- **后端**: Spring Boot 2.7+ / Spring Security / MyBatis Plus / Redis / WebSocket
- **前端**: Vue 3 + TypeScript + Element Plus + Pinia
- **数据库**: MySQL 8.0
- **消息队列**: RabbitMQ (可选)
- **部署**: Docker + Nginx

```java
// 竞价控制器
@Controller
public class BidController {

    @Autowired
    private BidService bidService;

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    @MessageMapping("/bid/{itemId}")
    @SendTo("/topic/bid/{itemId}")
    public BidResponse handleBid(@DestinationVariable Long itemId, 
                                BidRequest request, 
                                Principal principal) {
        // 验证用户身份
        String username = principal.getName();

        // 使用Redis分布式锁防止并发问题
        String lockKey = "item:lock:" + itemId;
        RLock lock = redissonClient.getLock(lockKey);

        try {
            if (lock.tryLock(3, 10, TimeUnit.SECONDS)) {
                // 处理出价逻辑
                BidResult result = bidService.processBid(itemId, username, request);

                // 广播出价信息
                return BidResponse.success(result);
            }
        } catch (Exception e) {
            log.error("Bid processing error", e);
        } finally {
            if (lock.isHeldByCurrentThread()) {
                lock.unlock();
            }
        }

        return BidResponse.fail("出价失败");
    }
}
```

#### 4.2.2 拍卖服务实现

```java
@Service
@Slf4j
public class AuctionServiceImpl implements AuctionService {

    @Autowired
    private AuctionItemMapper itemMapper;

    @Autowired
    private BidRecordMapper bidMapper;

    @Autowired
    private RedisTemplate<String, Object> redisTemplate;

    @Override
    @Transactional(rollbackFor = Exception.class)
    public BidResult processBid(Long itemId, String username, BidRequest request) {
        // 1. 获取拍品信息（优先从Redis获取）
        String cacheKey = "item:info:" + itemId;
        AuctionItem item = (AuctionItem) redisTemplate.opsForValue().get(cacheKey);

        if (item == null) {
            item = itemMapper.selectById(itemId);
            if (item != null) {
                redisTemplate.opsForValue().set(cacheKey, item, 1, TimeUnit.HOURS);
            }
        }

        // 2. 验证拍卖状态
        if (item.getStatus() != 3) {
            throw new BusinessException("拍卖未开始或已结束");
        }

        // 3. 验证出价
        BigDecimal bidAmount = request.getBidAmount();
        if (item.getCurrentPrice() != null) {
            if (bidAmount.compareTo(item.getCurrentPrice().add(item.getBidIncrement())) < 0) {
                throw new BusinessException("出价必须高于当前价加加价幅度");
            }
        } else {
            if (bidAmount.compareTo(item.getStartPrice()) < 0) {
                throw new BusinessException("出价不能低于起拍价");
            }
        }

        // 4. 保存出价记录
        BidRecord record = new BidRecord();
        record.setItemId(itemId);
        record.setUserId(getUserId(username));
        record.setBidAmount(bidAmount);
        record.setBidTime(new Date());
        bidMapper.insert(record);

        // 5. 更新拍品当前价格
        item.setCurrentPrice(bidAmount);
        itemMapper.updateById(item);

        // 6. 更新Redis缓存
        redisTemplate.opsForValue().set(cacheKey, item, 1, TimeUnit.HOURS);

        // 7. 添加到出价排行榜（Redis Sorted Set）
        String rankKey = "item:rank:" + itemId;
        redisTemplate.opsForZSet().add(rankKey, username, bidAmount.doubleValue());

        return BidResult.builder()
                .itemId(itemId)
                .currentPrice(bidAmount)
                .bidder(username)
                .bidTime(new Date())
                .build();
    }

    @Override
    @Scheduled(fixedDelay = 1000)
    public void checkAuctionEnd() {
        // 定时检查拍卖结束的拍品
        List<AuctionItem> endingItems = itemMapper.selectList(
                new LambdaQueryWrapper<AuctionItem>()
                        .eq(AuctionItem::getStatus, 3)
                        .le(AuctionItem::getEndTime, new Date())
        );

        for (AuctionItem item : endingItems) {
            endAuction(item.getId());
        }
    }

    @Override
    @Transactional
    public void endAuction(Long itemId) {
        // 1. 获取最高出价者
        BidRecord highestBid = bidMapper.selectOne(
                new LambdaQueryWrapper<BidRecord>()
                        .eq(BidRecord::getItemId, itemId)
                        .orderByDesc(BidRecord::getBidAmount)
                        .last("LIMIT 1")
        );

        // 2. 判断是否达到保留价
        AuctionItem item = itemMapper.selectById(itemId);

        if (highestBid != null && 
            highestBid.getBidAmount().compareTo(item.getReservePrice()) >= 0) {
            // 成交
            item.setStatus(4);
            highestBid.setIsWin(1);
            bidMapper.updateById(highestBid);

            // 生成订单
            createOrder(item, highestBid);
        } else {
            // 流拍
            item.setStatus(5);
        }

        itemMapper.updateById(item);
    }
}
```

#### 5.2.1 WebSocket连接管理

```typescript
// src/utils/websocket.ts
import SockJS from 'sockjs-client';
import { Client } from '@stomp/stompjs';
import { useUserStore } from '@/stores/user';

class WebSocketService {
  private stompClient: Client | null = null;
  private subscriptions: Map<string, any> = new Map();

  connect() {
    const userStore = useUserStore();
    const token = userStore.token;

    this.stompClient = new Client({
      webSocketFactory: () => new SockJS('http://localhost:8080/ws'),
      connectHeaders: {
        Authorization: `Bearer ${token}`
      },
      debug: (str) => {
        console.log('STOMP: ' + str);
      },
      reconnectDelay: 5000,
      heartbeatIncoming: 4000,
      heartbeatOutgoing: 4000
    });

    this.stompClient.onConnect = (frame) => {
      console.log('Connected: ' + frame);
    };

    this.stompClient.onStompError = (frame) => {
      console.error('Broker reported error: ' + frame.headers['message']);
    };

    this.stompClient.activate();
  }

  subscribeBidUpdate(itemId: number, callback: (data: any) => void) {
    if (!this.stompClient?.connected) {
      setTimeout(() => this.subscribeBidUpdate(itemId, callback), 1000);
      return;
    }

    const subscription = this.stompClient.subscribe(
      `/topic/bid/${itemId}`,
      (message) => {
        const bidData = JSON.parse(message.body);
        callback(bidData);
      }
    );

    this.subscriptions.set(`bid-${itemId}`, subscription);
  }

  sendBid(itemId: number, bidAmount: number) {
    if (!this.stompClient?.connected) {
      throw new Error('WebSocket not connected');
    }

    this.stompClient.publish({
      destination: `/app/bid/${itemId}`,
      body: JSON.stringify({ bidAmount })
    });
  }

  disconnect() {
    this.subscriptions.forEach(sub => sub.unsubscribe());
    this.subscriptions.clear();

    if (this.stompClient) {
      this.stompClient.deactivate();
    }
  }
}

export default new WebSocketService();
```

```vue
<script setup lang="ts">
import { ref, computed, onMounted, onUnmounted, watch } from 'vue';

const props = defineProps<{
  endTime: string | Date;
  label?: string;
}>();

const emit = defineEmits<{
  timeout: [];
}>();

const remaining = ref(0);
const timer = ref<number>();

const days = computed(() => Math.floor(remaining.value / 86400).toString().padStart(2, '0'));
const hours = computed(() => Math.floor((remaining.value % 86400) / 3600).toString().padStart(2, '0'));
const minutes = computed(() => Math.floor((remaining.value % 3600) / 60).toString().padStart(2, '0'));
const seconds = computed(() => Math.floor(remaining.value % 60).toString().padStart(2, '0'));

const isEnding = computed(() => remaining.value <= 300); // 5分钟内高亮显示

const calculateRemaining = () => {
  const end = new Date(props.endTime).getTime();
  const now = Date.now();
  const diff = Math.max(0, Math.floor((end - now) / 1000));

  remaining.value = diff;

  if (diff === 0) {
    emit('timeout');
    if (timer.value) {
      clearInterval(timer.value);
    }
  }

  return diff;
};

onMounted(() => {
  calculateRemaining();
  timer.value = setInterval(calculateRemaining, 1000);
});

onUnmounted(() => {
  if (timer.value) {
    clearInterval(timer.value);
  }
});

watch(() => props.endTime, () => {
  calculateRemaining();
});
</script>
```

```java
@Component
public class RateLimiter {

    private final RedisTemplate<String, Object> redisTemplate;

    // 出价频率限制
    public boolean checkBidRate(String userId, Long itemId) {
        String key = "rate:bid:" + userId + ":" + itemId;
        Long count = redisTemplate.opsForValue().increment(key);

        if (count == 1) {
            redisTemplate.expire(key, 3, TimeUnit.SECONDS);
        }

        return count <= 5; // 3秒内最多5次出价
    }

    // 接口访问限制
    public boolean checkApiRate(String ip, String api) {
        String key = "rate:api:" + api + ":" + ip;
        Long count = redisTemplate.opsForValue().increment(key);

        if (count == 1) {
            redisTemplate.expire(key, 60, TimeUnit.SECONDS);
        }

        return count <= 100; // 每分钟最多100次请求
    }
}
```
